---
title: "Companies in a Market"
api: "GET https://o2iren52nf.execute-api.eu-west-2.amazonaws.com/dev/markets/{market_id}/companies"
description: "This endpoint gets the companies in a given market."
---

### Path

<ParamField path="market_id" type="string" initialValue="1">
  The ID of the market whose companies you want to retrieve.
</ParamField>

### Query Parameters

<ParamField query="perPage" type="string" default="10" initialValue="10">
  Specifies the number of companies to retrieve per page.
</ParamField>

<ParamField query="page" type="string" default="1" initialValue="1">
  The page number to retrieve.
</ParamField>

### Response

<ResponseField name="id" type="string">
  The unique domain identifier for the company.
</ResponseField>

<ResponseField name="logoURL" type="string">
  The URL pointing to the company's logo.
</ResponseField>

<ResponseField name="name" type="string">
  The official name of the company.
</ResponseField>

<ResponseField name="description" type="string">
  A brief description of the company and its operations.
</ResponseField>

<ResponseField name="foundedYear" type="number">
  The year the company was founded.
</ResponseField>

<ResponseField name="totalFundingAmount" type="number|null">
  The total amount of funding received by the company. Null if not available.
</ResponseField>

<ResponseField name="hqAddress" type="string">
  The full address of the company's headquarters.
</ResponseField>

<ResponseField name="hqCity" type="string">
  The city where the company's headquarters is located.
</ResponseField>

<ResponseField name="hqState" type="string">
  The state or region of the company's headquarters.
</ResponseField>

<ResponseField name="hqCountry" type="string">
  The country where the company's headquarters is located.
</ResponseField>

<ResponseField name="size" type="string">
  The size or employee range of the company.
</ResponseField>

<ResponseField name="fundingRounds" type="number|null">
  The total number of funding rounds the company has gone through. Null if not available.
</ResponseField>

<ResponseField name="IPODate" type="date|null">
  The date when the company went public. Null if it hasn't or the date is not available.
</ResponseField>

<ResponseField name="revenue" type="string">
  The estimated revenue range of the company.
</ResponseField>

<ResponseField name="linkedInHandle" type="string">
  The company's LinkedIn profile handle.
</ResponseField>

<ResponseField name="twitterHandle" type="string">
  The company's Twitter handle.
</ResponseField>

<ResponseField name="tags" type="array">
  An array of tags or services associated with the company.
</ResponseField>

<ResponseField name="momentum" type="object">
  A score of how well the company is performing in this market, based on different factors.
  A score of 100 being the best.

  <Expandable title="momentum">

    <ResponseField name="esg" type="number">
      The company's Environmental, Social, and Governance (ESG) score.
    </ResponseField>

    <ResponseField name="overall" type="number">
      The company's general momentum score.
    </ResponseField>

    <ResponseField name="webTraffic" type="number">
      The company's web traffic score.
    </ResponseField>

    <ResponseField name="mediaAnalysis" type="number">
      The company's media analysis score.
    </ResponseField>

    <ResponseField name="revenueEstimates" type="number">
      The company's revenue estimates score.
    </ResponseField>

  </Expandable>

</ResponseField>

<ResponseField name="funding" type="object|null">
  Details about the company's funding. Null if not available.
</ResponseField>

<ResponseField name="entryPoint" type="string|null">
  The entry point or initial contact for the company. Null if not provided.
</ResponseField>

<ResponseField name="contactNumber" type="string">
  The primary contact number for the company.
</ResponseField>

<ResponseField name="contactEmail" type="string">
  The primary contact email for the company.
</ResponseField>
