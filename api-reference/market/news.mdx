---
title: "Market News"
api: "GET https://o2iren52nf.execute-api.eu-west-2.amazonaws.com/dev/markets/{market_id}/news"
description: "Retrieve a list of news articles and events specifically tailored for a given market, providing up-to-date insights and happenings related to that market."
---

### Path

<ParamField path="market_id" type="string" initialValue="1">
  This is the id of the market
</ParamField>

### Query

<ParamField query="searchTerm" type="string" initialValue="">
  Filter the news articles based on a specific keyword or phrase.
</ParamField>

<ParamField query="page" type="string" initialValue="1">
  The page, used to paginate through results.
</ParamField>

<ParamField query="perPage" type="string" initialValue="25">
  Number of news articles returned per page (Maximum 200).
</ParamField>


### Response

<ResponseField name="company_domain" type="string">
  The domain of the company the news article is about.
</ResponseField>

<ResponseField name="URL" type="string">
  The direct link to the news article.
</ResponseField>

<ResponseField name="title" type="string">
  The title of the news article.
</ResponseField>

<ResponseField name="snippet" type="string">
  A brief summary or extract from the article.
</ResponseField>

<ResponseField name="date_published" type="string">
  The date and time when the article was published in the format "YYYY-MM-DDTHH:MM:SS.sssZ".
</ResponseField>

<ResponseField name="source" type="string">
  The source or publisher of the news article.
</ResponseField>

<ResponseField name="source_logo_base64" type="string">
  A base64 encoded string representing the logo image of the news source.
</ResponseField>

<ResponseField name="text" type="string">
  The main content or body of the news article.
</ResponseField>
